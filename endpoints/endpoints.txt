# endpoints/endpoints.txt
# One backend per line. Lines starting with # are ignored.
# Format: http://IP:PORT   # Optional comment (provider, name, cost, etc.)
#
# Examples:

# Local Windows 11 + RTX 3070 Ti node (via Docker Desktop)
http://192.168.1.105:8000   # provider=local, name=Win11-3070Ti, cost=0

# Ubuntu server node
# http://10.0.0.50:8000     # provider=local, name=Ubuntu-vLLM, cost=0

# Akash decentralized deployment
# http://your-akash-deployment.example.com:8000   # provider=akash, name=Akash-Qwen, cost=0.3

# AIOZ Network (example)
# http://aioz-provider.example.com:8000   # provider=aioz, name=AIOZ-GPU-Node, cost=0.15

# Dynex (example)
# http://dynex-provider.example.com:8000   # provider=dynex, name=Dynex-GPU-Node, cost=0.12
